function scramble(str1, str2) {
    const str1Counts = [...str1].reduce(
        (counts, char) => (counts[char] = (counts[char] || 0) + 1, counts), {});
    const str2Counts = [...str2].reduce(
        (counts, char) => (counts[char] = (counts[char] || 0) + 1, counts), {});
    return Object.keys(str2Counts).every(
        (char) => str1Counts[char] >= str2Counts[char]);
}
